cmake_minimum_required(VERSION 3.10)
project(BallerEngine VERSION 1.0)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -lglfw -lGL -lX11 -lpthread -lXrandr -lXi -ldl")
include_directories(include)
link_directories(lib)
add_executable(BallerEngine src/GLFWTest.cpp src/VertexShader.vert src/FragmentShader.glsl dep/glad.c)
target_link_libraries(BallerEngine glm)
